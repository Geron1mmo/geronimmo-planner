WINDOWS:
  Double-click: D:\Geronimmo_Planner\RUN_GERONIMMO.bat
  Or run:       D:\Geronimmo_Planner\executables\windows\Geronimmo_Planner\geronimmo_planner.exe

ANDROID:
  Install: D:\Geronimmo_Planner\executables\android\Geronimmo_Planner.apk

REBUILD:
  cd D:\Geronimmo_Planner
  set JAVA_HOME=C:\Program Files\Android\Android Studio\jbr
  flutter build windows --release
  flutter build apk --release
